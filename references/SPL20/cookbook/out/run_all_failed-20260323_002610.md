        WHEN 'harmful' THEN
            COMMIT 'I cannot help with that request.' WITH
                status = 'blocked_harmful',
                gate   = 'keyword_prescreen'
        WHEN 'off_topic' THEN
            COMMIT 'That question is outside my scope. Please ask something relevant.' WITH
                status = 'blocked_off_topic',
                gate   = 'keyword_prescreen'
        OTHERWISE
            -- Continue to LLM classification
    END

    -- Gate 1b: LLM input content classification (nuanced cases keywords miss)
    GENERATE classify_input(@user_input, @test_context) INTO @input_class

    EVALUATE @input_class
        WHEN 'harmful' THEN
            COMMIT 'I cannot help with that request.' WITH
                status = 'blocked_harmful',
                gate   = 'llm_classifier'
        WHEN 'off_topic' THEN
            COMMIT 'That question is outside my scope. Please ask something relevant.' WITH
                status = 'blocked_off_topic',
                gate   = 'llm_classifier'
        OTHERWISE
            -- Input is acceptable, continue
    END

    -- Gate 2: PII detection (deterministic — no LLM)
    CALL detect_pii(@user_input) INTO @pii_report

    EVALUATE @pii_report
        WHEN 'pii_found' THEN
            -- Gate 2b: Redact PII (deterministic — no LLM)
            CALL redact_pii(@user_input) INTO @clean_input
        OTHERWISE
            SET @clean_input = @user_input
    END

    -- Gate 3: Generate response with safety-tuned prompt
    GENERATE safe_response(@clean_input) INTO @raw_response

    -- Gate 4: Output validation — check for PII leakage or harmful content
    GENERATE validate_output(@raw_response, @clean_input) INTO @output_check

    EVALUATE @output_check
        WHEN 'safe' THEN
            SET @safe_response = @raw_response
        WHEN 'contains_pii' THEN
            -- Deterministic PII redaction on the output too
            CALL redact_pii(@raw_response) INTO @safe_response
        WHEN 'hallucination' THEN
            GENERATE safe_response(@clean_input) INTO @safe_response
            RETRY WITH temperature = 0.1
        OTHERWISE
            SET @safe_response = 'I was unable to generate a safe response. Please rephrase your question.'
    END

    COMMIT @safe_response WITH
        status       = 'complete',
        input_class  = @input_class,
        pii_detected = @pii_report

EXCEPTION
    WHEN RefusalToAnswer THEN
        COMMIT 'This request cannot be processed.' WITH status = 'refused'
    WHEN HallucinationDetected THEN
        COMMIT 'Unable to provide a verified response.' WITH status = 'hallucination_blocked'
    WHEN BudgetExceeded THEN
        COMMIT @safe_response WITH status = 'budget_limit'
END

-- tool.request_approval() is a Python stub that auto-approves in dev mode

CREATE FUNCTION draft_report(topic TEXT, data TEXT) RETURNS TEXT AS $$
Write a concise 3-sentence executive summary report.
Be specific and factual. Avoid vague generalities.

Topic: {topic}
Source data: {data}
$$;

CREATE FUNCTION refine_report(draft TEXT, comments TEXT) RETURNS TEXT AS $$
Improve this executive summary based on reviewer comments.
Keep it to 3 sentences. Be specific and factual.

Original summary: {draft}
Reviewer comments: {comments}
$$;

WORKFLOW human_approval_gate
  INPUT:
    @topic TEXT DEFAULT 'Q1 Performance Review',
    @data  TEXT DEFAULT 'Revenue: $2.4M (+12% YoY). Churn: 3.2% (-0.8pp). NPS: 47 (+5).'
  OUTPUT:
    @final_report TEXT,
    @disposition  TEXT
DO
  -- Step 1: Generate draft
  GENERATE draft_report(@topic, @data) INTO @draft

  -- Step 2: Request approval (auto-approves in dev mode via stub tool)
  CALL request_approval(@draft) INTO @decision

  -- Step 3: Branch on decision
  EVALUATE @decision
    WHEN 'approved' THEN
      @final_report := @draft
      @disposition  := 'approved'
      COMMIT @final_report WITH status = 'approved'

    WHEN 'revise' THEN
      GENERATE refine_report(@draft, 'Make it more specific and concise.') INTO @final_report
      @disposition := 'revised'
      COMMIT @final_report WITH status = 'revised'

    OTHERWISE
      @final_report := @draft
      @disposition  := 'auto_approved'
      COMMIT @final_report WITH status = 'auto_approved'
  END

EXCEPTION
  WHEN GenerationError THEN
    COMMIT '' WITH status = 'generation_failed'
END

-- portability_test.spl
-- Pattern: One Recipe, Four Adapters — proving adapter portability
-- This file runs unchanged across --adapter ollama / claude_cli / openrouter / momagrid
-- Only the CLI flag changes. The workflow logic is identical on every backend.

CREATE FUNCTION write_summary(topic TEXT) RETURNS TEXT AS $$
Write a concise executive summary (exactly 3 sentences) about the following topic.
Be specific, factual, and direct. Avoid filler phrases.
Topic: {topic}
$$;

CREATE FUNCTION evaluate_summary(summary TEXT) RETURNS TEXT AS $$
Rate the quality of this executive summary on a scale from 0.0 to 1.0.
Criteria: specificity (no vague language), factual density, conciseness (no redundancy).
Summary: {summary}
Return ONLY a decimal number between 0.0 and 1.0. Nothing else.
$$;

CREATE FUNCTION refine_summary(summary TEXT, topic TEXT) RETURNS TEXT AS $$
Improve this executive summary. It was rated below threshold for quality.
Make it more specific, factual, and direct. Remove vague or redundant language.
Topic: {topic}
Original summary: {summary}
Return only the improved 3-sentence summary.
$$;

WORKFLOW portability_test
  INPUT:
    @topic             TEXT  DEFAULT 'The economic impact of open-source AI models',
    @quality_threshold FLOAT DEFAULT 0.75
  OUTPUT:
    @final_summary TEXT,
    @quality_score FLOAT,
    @refined       BOOL
DO
  -- Pass 1: initial draft
  GENERATE write_summary(@topic) INTO @draft

  -- Evaluate quality; parse the returned decimal string to a float
  GENERATE evaluate_summary(@draft) INTO @score_str
  CALL to_float(@score_str) INTO @quality_score

  EVALUATE @quality_score
    WHEN > @quality_threshold THEN
      @final_summary := @draft
      @refined := FALSE
      COMMIT @final_summary WITH quality = @quality_score, refined = 'no'

    OTHERWISE
      GENERATE refine_summary(@draft, @topic) INTO @final_summary
      @refined := TRUE
      COMMIT @final_summary WITH quality = @quality_score, refined = 'yes'
  END

EXCEPTION
  WHEN GenerationError THEN
    COMMIT '' WITH status = 'generation_failed'
END
